#### 4. `grok_coder/tokenizer.py` (Custom Code-Optimized Byte-Level BPE)
```python
from tokenizers import Tokenizer, models, trainers, pre_tokenizers, normalizers
from tokenizers.processors import TemplateProcessing
import json
from pathlib import Path

class GrokCoderTokenizer:
    def __init__(self, vocab_size: int = 32768):
        self.vocab_size = vocab_size
        self.tokenizer = None
        self.path = Path("grok_coder_tokenizer.json")

    def train(self, corpus: list[str]):
        """Train byte-level BPE on code-heavy corpus (byte_fallback + code patterns)."""
        tok = Tokenizer(models.BPE(byte_fallback=True))
        tok.normalizer = normalizers.NFKC()
        tok.pre_tokenizer = pre_tokenizers.ByteLevel(add_prefix_space=False)

        trainer = trainers.BpeTrainer(
            vocab_size=self.vocab_size,
            min_frequency=2,
            special_tokens=[
                "<|endoftext|>", "<|fim_prefix|>", "<|fim_middle|>", "<|fim_suffix|>",
                "<|python|>", "<|test_pass|>", "<|test_fail|>",
                "[PAD]", "[UNK]"
            ],
            continuing_subword_prefix="▁",
        )

        tok.train_from_iterator(corpus, trainer=trainer)
        tok.post_processor = TemplateProcessing(
            single="$A <|endoftext|>",
            pair="$A $B <|endoftext|>",
            special_tokens=[("<|endoftext|>", tok.token_to_id("<|endoftext|>"))]
        )
        self.tokenizer = tok
        self.save()

    def load(self):
        if self.path.exists():
            self.tokenizer = Tokenizer.from_file(str(self.path))
            return True
        return False

    def save(self):
        self.tokenizer.save(str(self.path))

    def encode(self, text: str):
        return self.tokenizer.encode(text).ids

    def decode(self, ids):
        return self.tokenizer.decode(ids)

    def get_vocab_size(self):
        return self.tokenizer.get_vocab_size()

# Sample code corpus for instant training (expand with The Stack later)
SAMPLE_CODE = [
    "def fibonacci(n: int) -> int:\n    if n <= 1:\n        return n\n    return fibonacci(n-1) + fibonacci(n-2)",
    "class TestCase:\n    def test_fib(self):\n        assert fibonacci(5) == 5",
    "import numpy as np\nx = np.array([1,2,3])",
    # ... add more real code in production
]

def build_tokenizer():
    tk = GrokCoderTokenizer()
    if not tk.load():
        print("Training custom Grok Coder tokenizer on code corpus...")
        tk.train(SAMPLE_CODE)
        print("Tokenizer ready.")
    return tk
```

#### 5. `grok_coder/sandbox.py` (Verifiable Execution Engine)
```python
import subprocess
import tempfile
import time
from typing import Tuple

class CodeSandbox:
    @staticmethod
    def execute_python(code: str, timeout: int = 10) -> Tuple[bool, str, str]:
        """Run code in isolated subprocess with timeout. Returns (success, stdout, stderr)."""
        with tempfile.NamedTemporaryFile(mode="w", suffix=".py", delete=False) as f:
            f.write(code)
            f_path = f.name

        try:
            result = subprocess.run(
                ["python", f_path],
                capture_output=True,
                text=True,
                timeout=timeout
            )
            success = result.returncode == 0
            return success, result.stdout.strip(), result.stderr.strip()
        except subprocess.TimeoutExpired:
            return False, "", "Execution timeout"
        except Exception as e:
            return False, "", str(e)
        finally:
            try:
                Path(f_path).unlink(missing_ok=True)
            except:
                pass
```

#### 6. `grok_coder/engine.py` (The Full Agentic Grok Coder Engine)
```python
from transformers import AutoModelForCausalLM, AutoTokenizer, BitsAndBytesConfig
import torch
from rich.console import Console
from rich.panel import Panel
from .tokenizer import build_tokenizer
from .sandbox import CodeSandbox
import re

console = Console()

class GrokCoderEngine:
    def __init__(self, model_name: str = "Qwen/Qwen2.5-Coder-1.5B-Instruct"):
        self.tokenizer = build_tokenizer()  # Our custom one
        self.hf_tokenizer = AutoTokenizer.from_pretrained(model_name, trust_remote_code=True)

        # 4-bit quantization for local speed
        quantization_config = BitsAndBytesConfig(
            load_in_4bit=True,
            bnb_4bit_compute_dtype=torch.bfloat16,
            bnb_4bit_use_double_quant=True,
            bnb_4bit_quant_type="nf4"
        )

        self.model = AutoModelForCausalLM.from_pretrained(
            model_name,
            quantization_config=quantization_config,
            device_map="auto",
            trust_remote_code=True
        )
        self.sandbox = CodeSandbox()
        console.print("[bold green]Grok Coder engine loaded[/bold green]")

    def generate(self, prompt: str, max_new_tokens: int = 1024) -> str:
        inputs = self.hf_tokenizer(prompt, return_tensors="pt").to(self.model.device)
        outputs = self.model.generate(
            **inputs,
            max_new_tokens=max_new_tokens,
            temperature=0.7,
            top_p=0.95,
            do_sample=True,
            eos_token_id=self.hf_tokenizer.eos_token_id
        )
        return self.hf_tokenizer.decode(outputs[0], skip_special_tokens=True)

    def agentic_code(self, task: str, language: str = "python", max_iterations: int = 5) -> str:
        """Full Grok Coder loop: Plan → Generate → Test → Self-correct."""
        console.print(Panel(f"[bold]Task:[/bold] {task}", title="Grok Coder Agentic Loop"))

        # Step 1: Plan
        plan_prompt = f"<|fim_prefix|>Task: {task}\nLanguage: {language}\nPlan step-by-step:<|fim_middle|>"
        plan = self.generate(plan_prompt, max_new_tokens=256)
        console.print(f"[dim]Plan:[/dim] {plan[:300]}...")

        # Step 2-5: Iterative generate + verify
        code = ""
        for i in range(max_iterations):
            # Generate with FIM + our custom tokenizer awareness
            gen_prompt = f"<|fim_prefix|>Task: {task}\nLanguage: {language}\nPlan: {plan}\nCode:\n{code}<|fim_middle|>"
            raw = self.generate(gen_prompt)
            # Extract only the generated code block
            match = re.search(r"```" + language + r"\n(.*?)```", raw, re.DOTALL)
            code = match.group(1).strip() if match else raw.strip()

            console.print(f"[yellow]Iteration {i+1} generated code[/yellow]")

            # Execute + verify
            success, stdout, stderr = self.sandbox.execute_python(code)
            if success:
                console.print("[bold green]✅ Tests passed! Final code:[/bold green]")
                console.print(code)
                return code
            else:
                console.print(f"[red]❌ Failed: {stderr}[/red]")
                # Self-correction prompt
                fix_prompt = f"Previous code failed:\n{code}\nError: {stderr}\nFix it completely:\n"
                code = self.generate(fix_prompt, max_new_tokens=512)

        console.print("[red]Max iterations reached. Best attempt:[/red]")
        return code

# Singleton for CLI
_engine = None
def get_engine():
    global _engine
    if _engine is None:
        _engine = GrokCoderEngine()
    return _engine
```

#### 7. `cli.py` (Run it!)
```python
import typer
from grok_coder.engine import get_engine
from rich.console import Console

app = typer.Typer()
console = Console()

@app.command()
def code(task: str):
    """Run Grok Coder on any coding task."""
    engine = get_engine()
    result = engine.agentic_code(task)
    console.print("\n[bold]Final verified code:[/bold]")
    console.print(result)

if __name__ == "__main__":
    app()
```
