grok_coder/confidence.py

""" Objective Confidence Engine — v4 core feature. Replaces the self-reported float with a ConfidenceScore computed from five independent, measurable signals. """ from __future__ import annotations import re import statistics from dataclasses import dataclass, field from typing import Optional @dataclass class ConfidenceScore: """ All sub-scores are normalised to 0.0–1.0 (higher = better). `overall` is a weighted harmonic mean so a single zero drags the result down. """ # Individual signal scores coverage: float = 0.0 # Test coverage % / 100 complexity: float = 0.0 # Radon MI normalised; 1.0 = perfectly simple static_score: float = 0.0 # 1.0 = no linter issues verifier_consensus: float = 0.0 # 1.0 = verifier agrees code is correct test_stability: float = 0.0 # 1.0 = tests passed all reruns without flaking # Weights (must sum to 1.0) _WEIGHTS: dict = field(default_factory=lambda: { "coverage": 0.30, "complexity": 0.10, "static_score": 0.20, "verifier_consensus": 0.20, "test_stability": 0.20, }, repr=False) @property def overall(self) -> float: """ Weighted harmonic mean — punishes zeros harshly. Falls back to weighted arithmetic mean if any weight maps to zero score. """ weights = self._WEIGHTS scores = { "coverage": self.coverage, "complexity": self.complexity, "static_score": self.static_score, "verifier_consensus": self.verifier_consensus, "test_stability": self.test_stability, } # Harmonic mean: 1 / sum(w_i / s_i) try: harmonic = 1.0 / sum( weights[k] / max(scores[k], 1e-6) for k in scores ) return round(min(harmonic, 1.0), 4) except ZeroDivisionError: arithmetic = sum(weights[k] * scores[k] for k in scores) return round(min(arithmetic, 1.0), 4) @property def grade(self) -> str: o = self.overall if o >= 0.90: return "A" if o >= 0.75: return "B" if o >= 0.60: return "C" if o >= 0.40: return "D" return "F" def console_summary(self) -> str: bar = "█" * int(self.overall * 20) + "░" * (20 - int(self.overall * 20)) return ( f"[{bar}] {self.overall:.2f} ({self.grade}) " f"cov={self.coverage:.0%} " f"cmplx={self.complexity:.0%} " f"lint={self.static_score:.0%} " f"verif={self.verifier_consensus:.0%} " f"stable={self.test_stability:.0%}" ) def to_dict(self) -> dict: return { "overall": self.overall, "grade": self.grade, "coverage": self.coverage, "complexity": self.complexity, "static_score": self.static_score, "verifier_consensus": self.verifier_consensus, "test_stability": self.test_stability, } class ConfidenceEngine: """ Computes a ConfidenceScore from sandbox results and static analysis output. """ Code-optimised byte-level BPE tokenizer. Scoped to: offline analysis and LoRA fine-tuning preparation. NOT used during inference (Qwen2.5-Coder HF tokenizer handles that). Unchanged from v3. """ from __future__ import annotations import logging from pathlib import Path from typing import Optional logger = logging.getLogger(__name__) class GrokCoderTokenizer: SPECIAL_TOKENS = [ "<|endoftext|>", "<|fim_prefix|>", "<|fim_middle|>", "<|fim_suffix|>", "<|python|>", "<|javascript|>", "<|typescript|>", "<|rust|>", "<|go|>", "<|test_pass|>", "<|test_fail|>", "[PAD]", "[UNK]", ] def __init__(self, vocab_size: int = 32768, save_path: str = "grok_coder_tokenizer.json"): self.vocab_size = vocab_size self.tokenizer = None self.path = Path(save_path) def train(self, corpus: list[str]) -> None: from tokenizers import Tokenizer, models, trainers, pre_tokenizers, normalizers from tokenizers.processors import TemplateProcessing if len(corpus) < 50: logger.warning("Corpus has fewer than 50 examples — tokenizer quality will be poor.") tok = Tokenizer(models.BPE(byte_fallback=True)) tok.normalizer = normalizers.NFKC() tok.pre_tokenizer = pre_tokenizers.ByteLevel(add_prefix_space=False) trainer = trainers.BpeTrainer( vocab_size=self.vocab_size, min_frequency=2, special_tokens=self.SPECIAL_TOKENS, ) tok.train_from_iterator(corpus, trainer=trainer) tok.post_processor = TemplateProcessing( single="$A <|endoftext|>", pair="$A $B <|endoftext|>", special_tokens=[("<|endoftext|>", tok.token_to_id("<|endoftext|>"))], ) self.tokenizer = tok self.save() logger.info("Tokenizer trained. Vocab size: %d", tok.get_vocab_size()) def load(self) -> bool: from tokenizers import Tokenizer if self.path.exists(): self.tokenizer = Tokenizer.from_file(str(self.path)) return True return False def save(self) -> None: if self.tokenizer: self.tokenizer.save(str(self.path)) def encode(self, text: str) -> list[int]: assert self.tokenizer, "Tokenizer not loaded or trained." return self.tokenizer.encode(text).ids def decode(self, ids: list[int]) -> str: assert self.tokenizer, "Tokenizer not loaded or trained." return self.tokenizer.decode(ids)

# Grok Coder v4 — Universal Verifiable Coding Engine A local, multi-language, self-verifying agentic coding engine with **objective confidence scoring**, Docker sandboxing, property-based testing, and an optional FastAPI server mode. ## Architecture Axioms 1. **Correctness is measurable** — ConfidenceScore is computed from real signals, not model self-reports. 2. **Security is language-agnostic** — the sandbox and AST scanner are the source of truth. 3. **Extensibility is zero-boilerplate** — adding a new language takes < 200 lines + one Docker image. --- ## Requirements - Python 3.11+ - Docker (recommended — language-specific fallback available for Python) - CUDA GPU recommended (CPU fallback with estimated timing) - 4+ GB VRAM for 1.5B model (auto-selected); 8+ GB for 3B; 20+ GB for 7B --- ## Quick Start ```bash # 1. Install dependencies pip install -r requirements.txt # 2. Build sandbox images (one-time) docker build -f Dockerfile.sandbox-python -t grok-sandbox-python . docker build -f Dockerfile.sandbox-node -t grok-sandbox-node . docker build -f Dockerfile.sandbox-rust -t grok-sandbox-rust . docker build -f Dockerfile.sandbox-go -t grok-sandbox-go . # 3. Run tasks python cli.py "Implement a thread-safe LRU cache" --language python python cli.py "Build a radix sort" --language rust python cli.py "Debounce utility" --language javascript python cli.py "Binary search" --language go # 4. Higher coverage threshold python cli.py "Fibonacci with edge cases" --min-coverage 90 # 5. Start server mode python cli.py --serve # → http://localhost:8000/docs

Adding a New Language

Create grok_coder/languages/mylang.py implementing LanguageHandler ABC (~150 lines).

Add a Docker image Dockerfile.sandbox-mylang.

Register in grok_coder/languages/__init__.py (~3 lines).

Done. The agentic loop, confidence engine, and server mode work automatically.

Known Limitations

restrictedpython fallback is not a true sandbox (Python only; local dev use).

The verifier model adds ~2–3 s per iteration on 8 GB VRAM.

Rust sandbox image is larger (~500 MB) due to the Rust toolchain.

The singleton get_engine() serialises requests; create per-worker instances for production scale.
