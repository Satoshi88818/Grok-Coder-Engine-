grok_coder/structured_output.py

""" Structured output via instructor + Pydantic. This module replaces heuristic regex extraction with guaranteed structured generation. Works with: - HuggingFace local models (via a thin OpenAI-compatible shim) - vLLM / TGI backends (native OpenAI-compatible API) On failure (model too small, backend mismatch), falls back to the legacy extractor so the agentic loop never hard-crashes. """ from __future__ import annotations import logging from typing import Optional import instructor from openai import OpenAI from pydantic import BaseModel, Field, field_validator from tenacity import retry, stop_after_attempt, wait_exponential log = logging.getLogger(__name__) class GeneratedCode(BaseModel): """Structured output contract for a single code generation turn.""" reasoning: str = Field( description="Brief chain-of-thought explaining the approach (not shown to user)." ) implementation: str = Field( description="Complete implementation source code, no markdown fences." ) tests: str = Field( description="Complete test file source code, no markdown fences." ) explanation: str = Field( description="User-facing explanation of the implementation (1–3 sentences)." ) @field_validator("implementation", "tests") @classmethod def strip_fences(cls, v: str) -> str: """Remove accidental markdown code fences.""" lines = v.strip().splitlines() if lines and lines[0].startswith("```"): lines = lines[1:] if lines and lines[-1].strip() == "```": lines = lines[:-1] return "\n".join(lines).strip() class StructuredOutputClient: """ Wraps an OpenAI-compatible client with instructor for guaranteed structured output with Pydantic validation and retry on failure. """ def __init__(self, base_url: str, api_key: str = "local", model: str = "default"): raw_client = OpenAI(base_url=base_url, api_key=api_key) self._client = instructor.from_openai(raw_client, mode=instructor.Mode.JSON) self._model = model @retry( stop=stop_after_attempt(3), wait=wait_exponential(multiplier=1, min=1, max=10), reraise=True, ) def generate( self, system_prompt: str, user_prompt: str, temperature: float = 0.2, max_tokens: int = 2048, ) -> Optional[GeneratedCode]: """ Generate structured code output. Returns None only if all retries fail. The agentic loop should fall back to legacy extraction in that case. """ try: result = self._client.chat.completions.create( model=self._model, response_model=GeneratedCode, messages=[ {"role": "system", "content": system_prompt}, {"role": "user", "content": user_prompt}, ], temperature=temperature, max_tokens=max_tokens, ) return result except Exception as exc: log.warning("Structured output generation failed: %s", exc) raise # tenacity will retry @staticmethod def fallback_available() -> bool: """Check whether the legacy extractor fallback is importable.""" try: import grok_coder.extractor # noqa: F401 return True except ImportError: return False 

grok_coder/model_backend.py

""" Backend abstraction layer for Grok Coder v5. Supports three backends: - huggingface: local transformers pipeline (default, no network required) - vllm: vLLM OpenAI-compatible HTTP server - tgi: HuggingFace Text Generation Inference HTTP server The engine calls generate() without caring which backend is active. Prompt caching is enabled by default for vLLM/TGI (they handle it server-side). """ from __future__ import annotations import logging from abc import ABC, abstractmethod from typing import Any, Dict, Iterator, List, Optional log = logging.getLogger(__name__) class BaseBackend(ABC): """Common interface all backends must satisfy.""" @abstractmethod def generate( self, messages: List[Dict[str, str]], *, max_tokens: int = 2048, temperature: float = 0.2, stream: bool = False, ) -> str | Iterator[str]: """ Generate a completion for the provided message list. When stream=True, yield token strings incrementally. """ ... @abstractmethod def health(self) -> bool: """Return True if the backend is reachable and ready.""" ... class HuggingFaceBackend(BaseBackend): """Local transformers pipeline backend.""" def __init__(self, model_name: str, load_in_4bit: bool = True, cpu_offload: bool = False): import torch from transformers import AutoModelForCausalLM, AutoTokenizer, pipeline, BitsAndBytesConfig log.info("Loading HuggingFace model: %s (4bit=%s)", model_name, load_in_4bit) quantization_config = None if load_in_4bit: try: quantization_config = BitsAndBytesConfig(load_in_4bit=True) except Exception: log.warning("4-bit quantization unavailable; trying 8-bit fallback.") try: quantization_config = BitsAndBytesConfig(load_in_8bit=True) except Exception: log.warning("8-bit also unavailable; loading in full precision.") model = AutoModelForCausalLM.from_pretrained( model_name, quantization_config=quantization_config, device_map="auto" if not cpu_offload else "cpu", torch_dtype=torch.float16 if not cpu_offload else torch.float32, ) tokenizer = AutoTokenizer.from_pretrained(model_name) self._pipe = pipeline( "text-generation", model=model, tokenizer=tokenizer, ) self._tokenizer = tokenizer def generate( self, messages: List[Dict[str, str]], *, max_tokens: int = 2048, temperature: float = 0.2, stream: bool = False, ) -> str: prompt = self._tokenizer.apply_chat_template( messages, tokenize=False, add_generation_prompt=True ) result = self._pipe( prompt, max_new_tokens=max_tokens, temperature=temperature, do_sample=temperature > 0, return_full_text=False, ) return result[0]["generated_text"] def health(self) -> bool: return self._pipe is not None class OpenAICompatibleBackend(BaseBackend): """ Generic backend for any OpenAI-compatible API server. Used for both vLLM and TGI. """ def __init__(self, base_url: str, model: str, backend_type: str = "vllm"): from openai import OpenAI self._client = OpenAI(base_url=base_url, api_key="local") self._model = model self._backend_type = backend_type log.info("Using %s backend at %s (model=%s)", backend_type, base_url, model) def generate( self, messages: List[Dict[str, str]], *, max_tokens: int = 2048, temperature: float = 0.2, stream: bool = False, ) -> str | Iterator[str]: if stream: return self._stream(messages, max_tokens=max_tokens, temperature=temperature) response = self._client.chat.completions.create( model=self._model, messages=messages, max_tokens=max_tokens, temperature=temperature, ) return response.choices[0].message.content def _stream( self, messages: List[Dict[str, str]], max_tokens: int, temperature: float, ) -> Iterator[str]: with self._client.chat.completions.stream( model=self._model, messages=messages, max_tokens=max_tokens, temperature=temperature, ) as stream: for text in stream.text_stream: yield text def health(self) -> bool: try: models = self._client.models.list() return len(models.data) > 0 except Exception: return False def build_backend(config: Any) -> BaseBackend: """Factory: instantiate the correct backend from config.""" if config.backend == "huggingface": return HuggingFaceBackend( model_name=config.model_name, load_in_4bit=config.load_in_4bit, cpu_offload=config.cpu_offload, ) if config.backend in ("vllm", "tgi"): if not config.backend_url: raise ValueError( f"GROK_BACKEND_URL must be set when using backend='{config.backend}'" ) return OpenAICompatibleBackend( base_url=config.backend_url, model=config.model_name, backend_type=config.backend, ) raise ValueError(f"Unknown backend: {config.backend!r}") 

grok_coder/model_router.py

""" Resource-aware model and backend selection for Grok Coder v5. Selection priority: 1. Explicit GROK_MODEL env var → use as-is 2. vLLM / TGI backend configured → skip local VRAM check 3. Local HuggingFace → VRAM tier auto-selection """ from __future__ import annotations import logging import os from typing import Optional, Tuple import psutil log = logging.getLogger(__name__) # VRAM tier thresholds (GiB) _TIERS: list[Tuple[float, str]] = [ (20.0, "Qwen/Qwen2.5-Coder-7B-Instruct"), (8.0, "Qwen/Qwen2.5-Coder-3B-Instruct"), (0.0, "Qwen/Qwen2.5-Coder-1.5B-Instruct"), ] def _available_vram_gib() -> float: try: import torch if torch.cuda.is_available(): return torch.cuda.get_device_properties(0).total_memory / (1024 ** 3) except Exception: pass return 0.0 def select_model(explicit_model: Optional[str] = None, backend: str = "huggingface") -> str: if explicit_model and explicit_model != "Qwen/Qwen2.5-Coder-1.5B-Instruct": log.info("Model explicitly set: %s", explicit_model) return explicit_model if backend in ("vllm", "tgi"): chosen = explicit_model or "Qwen/Qwen2.5-Coder-1.5B-Instruct" log.info("Remote backend (%s) — using model: %s", backend, chosen) return chosen vram = _available_vram_gib() log.info("Detected VRAM: %.1f GiB", vram) for threshold, model in _TIERS: if vram >= threshold: log.info("Auto-selected model: %s (VRAM >= %.1f GiB)", model, threshold) return model return _TIERS[-1][1] # minimum-footprint fallback

grok_coder/languages/base.py

""" Abstract base class for language handlers. v5 change: each handler now exposes `confidence_weights` to allow per-language tuning of the ConfidenceScore signals. """ from __future__ import annotations import abc from dataclasses import dataclass, field from pathlib import Path from typing import Any, Optional @dataclass class SandboxResult: success: bool stdout: str stderr: str exit_code: int coverage_pct: Optional[float] = None coverage_report: Optional[str] = None elapsed_s: float = 0.0 @dataclass class StaticResult: passed: bool issues: list[str] = field(default_factory=list) complexity_score: Optional[float] = None @dataclass class ConfidenceWeights: """ Per-language confidence signal weights. """ Verifier model — v5. Changes from v4: - Returns a scored rubric float (0–1) instead of binary yes/no - Rubric evaluates: correctness, completeness, idiomatic style (1–5 each) - Falls back to 0.5 (neutral) on any error """ from __future__ import annotations import logging import re from typing import Optional log = logging.getLogger(__name__) _RUBRIC_PROMPT = """\ You are a senior code reviewer. Evaluate the following code and tests on three dimensions. For each dimension, respond with a score from 1 (poor) to 5 (excellent). DIMENSIONS: 1. Correctness: Does the implementation correctly solve the stated task? 2. Completeness: Are edge cases handled? Are tests thorough? 3. Idiomaticity: Is the code idiomatic for the language (style, naming, patterns)? TASK: {task} CODE: {code} TESTS: {tests} Respond ONLY in this exact format (no other text): Correctness: <1-5> Completeness: <1-5> Idiomaticity: <1-5> """ class VerifierModel: def __init__(self, backend, model_name: str): self._backend = backend self._model_name = model_name def score( self, task: str, code: str, tests: str, language: str = "python", ) -> Optional[float]: """ Returns a normalised float in [0, 1] based on the rubric, or None if the verifier fails. """ prompt = _RUBRIC_PROMPT.format(task=task, code=code[:2000], tests=tests[:1000]) try: messages = [{"role": "user", "content": prompt}] raw = self._backend.generate( messages, max_tokens=64, temperature=0.0 ) return self._parse_rubric(raw) except Exception as exc: log.warning("Verifier failed: %s — using neutral score 0.5", exc) return None # engine will use 0.5 neutral @staticmethod def _parse_rubric(raw: str) -> Optional[float]: scores = [] for dimension in ("Correctness", "Completeness", "Idiomaticity"): match = re.search(rf"{dimension}:\s*([1-5])", raw, re.IGNORECASE) if match: scores.append(int(match.group(1))) if len(scores) < 2: log.warning("Could not parse verifier rubric from: %r", raw[:200]) return None # Normalise: mean of available scores, then scale to 0–1 mean = sum(scores) / len(scores) return (mean - 1) / 4.0 # maps [1,5] → [0.0, 1.0]

# Grok Coder v5 — Universal Verifiable Coding Engine A local, multi-language, self-verifying agentic coding engine with **objective confidence scoring**, Docker sandboxing, first-class TypeScript support, tree-sitter multi-language AST scanning, instructor-based structured output, and an optional FastAPI server mode. ## What's New in v5 | Feature | v4 | v5 | |---|---|---| | TypeScript | Aliased to JS | First-class handler (ts-jest, tsc, @typescript-eslint) | | AST scanning | Python only | All 5 languages via tree-sitter | | Code extraction | Heuristic regex | instructor + Pydantic (fallback to regex) | | Verifier | Binary yes/no | Rubric float 0–1 (correctness + completeness + idiomaticity) | | Confidence weights | Uniform across languages | Per-language (JS/TS/Rust elevated stability) | | Sandbox security | USER sandbox only | + seccomp profile + --cap-drop=ALL | | Backend | HuggingFace only | HuggingFace + vLLM + TGI (OpenAI-compatible) | | Mutation testing | — | mutmut (Python, when coverage ≥ 85%) | | CLI flags | Limited | --temperature, --max-tokens, --backend, --mutation, --open-report | | Server mode | Singleton engine | Per-request engine instances | | Test suite | Unit + golden | + WebSocket/REST integration tests | ## Architecture Axioms 1. **Correctness is measurable** — ConfidenceScore is computed from real signals, not model self-reports. 2. **Security is language-agnostic** — tree-sitter scanner + seccomp sandbox are the source of truth. 3. **Extensibility is zero-boilerplate** — adding a new language takes < 200 lines + one Docker image. 4. **Structure over heuristics** — instructor guarantees code/tests are parsed correctly from the model. ## Requirements - Python 3.11+ - Docker (recommended) - CUDA GPU recommended; CPU fallback available - 4+ GB VRAM for 1.5B model; 8+ GB for 3B; 20+ GB for 7B - For vLLM/TGI mode: a running vLLM or TGI server ## Quick Start ```bash # 1. Create grok_coder/languages/mylang.py implementing LanguageHandler ABC (~150 lines).

Known Limitations

restrictedpython fallback is not a true sandbox (Python only; local dev use).

Mutation testing (mutmut) is slow; only enabled explicitly and only for Python.

The TypeScript handler requires npm install -g typescript ts-jest jest @types/jest in the sandbox image.

vLLM structured output requires --guided-decoding-backend outlines in the vLLM server config.

Rust sandbox image is ~500 MB due to the Rust toolchain.
